表格组件

<template>
<nf-table
   ref="table"
   :config="tableConfig"
   :table-param="tableParam"
   @reset="resetQuery"
   @search="search"
   @multipleDelHandler="resetQuery"
   @deleteRowHandler="resetQuery"
   @addDept="resetQuery"
   @editDept="resetQuery">
</nf-table> 
</template>
<script>
    export default {
        mounted(){
            this.$refs.table.tableData = [
                {deptCode:'1001',deptName:'例子1',deptTelPerson:'例先生',deptTel:'15855888855',deptDescription:'描述例子',remark:'我是备注'},
                {deptCode:'1002',deptName:'例子2',deptTelPerson:'例先生2',deptTel:'15855888856',deptDescription:'描述例子2',remark:'我是备注2'},
            ]
        },      
        data() {
              return {
                tableConfig: {
                  title: "table标题",
                  isNumber: false,
                  isSelect: true,
                  rowKey: "baseId",
                  api: "example/selectPageListDepartment",
                  query: {
                    perCode: "",
                    config: {
                      queryCol: 1,
                      defaultQuery: {type: "input", prop: "memberCode", placeholder: "请输入属性代码"},
                      query: [
                        {type: "input", label: "属性名称", prop: "memberName", placeholder: "请输入属性名称"},
                        {type: "input", label: "属性类型", prop: "memberType", placeholder: "请输入属性类型"},
                        {type: "input", label: "作用域", prop: "memberScope", placeholder: "请输入属性作用域"}
                      ]
                    }
                  },
                  columns: [
                    {label: "科室编码", prop: "deptCode"},
                    {label: "科室名称", prop: "deptName"},
                    {label: "科室联系人", prop: "deptTelPerson"},
                    {label: "科室电话", prop: "deptTel"},
                    {label: "科室描述", prop: "deptDescription"},
                    {label: "备注", prop: "remark"}
                  ],
                  tableOperation: [
                    {
                      name: "增加",
                      type: "primary",
                      plain: true,
                      // operationType: this.$nfConst.TABLE_OPERATION_TYPE.ADD,
                      emitName: "增加事件",
                      // perCode: "增加按钮权限码",
                    },
                    {
                      name: "删除",
                      type: "danger",
                      plain: true,
                      emitName: "multipleDelHandler",
                      // operationType: this.$nfConst.TABLE_OPERATION_TYPE.DELETE,
                      api: "api_dept/deleteListDepartment",
                      // perCode: "btn-dept-delete",
                    },
                  ],
                  columnOperation: {
                    width: "120",
                    label: "操作",
                    operations: [
                      {
                        name: "编辑",
                        type: "primary",
                        emitName: "editDept",
                        // operationType: this.$nfConst.TABLE_COLUMN_OPERATION_TYPE.DATA_OBJ,
                        // perCode: "btn-dept-edit",
                      },
                      {
                        name: "删除",
                        type: "danger",
                        emitName: "multipleData",
                        // operationType: this.$nfConst.TABLE_COLUMN_OPERATION_TYPE.DELETE,
                        api: "api_dept/deleteOneDepartment",
                        // perCode: "btn-dept-delete",
                      },
                    ],
                  },
                  pageConfig: {
                    pageCount: 5,
                    pageSizes: [15, 30, 50, 100],
                  },
                },
                tableParam: {deptOrgBaseId: "606486b2e2076e4734b449cd"}
              }
        },
        methods: {
              resetQuery(callback) {
                this.tableParam.namespace = 'aa';
                callback(true);
              },
              search(callback) {
                callback(true);
              },
              addDept() {
                this.tableConfig.title = "ddd";
              }
        }
    }
</script>
