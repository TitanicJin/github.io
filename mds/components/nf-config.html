# 配置

接收五个参数

httpConfig 用于配置request的各个参数
apis 用于将项目中的所有请求集中管理
routers 项目可以跳转的路由
stores 项目的状态
tabs 项目的标签页

import apis from './apis/index'
import httpConfig from './config/http-config'
import routers from "./config/router-config"
import stores from "./store/index"
import tabs from "./config/tabs-config"
//创建框架的初始化配置
let nfConfig = {
     httpConfig: httpConfig,
     routers: routers,
     stores: stores,
     tabs: tabs,
     apis: apis
}

# httpConfig

本节将介绍框架初始化中的网络请求配置

export default {
  //请求可用类型
  method: 'get,post',
  // 基础url前缀，可根据不同环境设置不同的路径
  baseURL: 'http://localhost:8521/api/',
  // 请求头信息
  headers: {
    'Content-Type': 'application/json;charset=UTF-8'
  },
  // 请求默认参数
  data: {},
  // 设置超时时间
  timeout: 18000,
  // 携带凭证
  withCredentials: true,
  // 响应数据类型
  responseType: 'json'
}

# apis

本节将介绍框架初始化中的api配置

定义一个api

import UtilMessage from "newlife-components/src/utils/js/UtilMessage";
const baseUrl = 'xx/xx';
export default (httpInstance) => {
  return {
    apiExample1(param) {
      return httpInstance.post(baseUrl + "/example1", param);
    },
    apiExample2(param) {
      return UtilMessage.confirm("确定要解除该角色吗？").then(async () => {
        return httpInstance.post(baseUrl + "/example1", param);
      });
    },
  }
}

暴露所有api

import apis from "./apis/index";
export default {
  api_xxx
}

# routers

本节将介绍框架初始化中的路由配置

//路由页面
import OrgMain from "../views/OrgMain";
//配置路由
const routers = {
  main: {
    name: OrgMain.name,
    component: OrgMain,
    path: "/nf-main",
    meta: {needToken: true},
  }
};
export default routers
参数树 说明 类型 是否必填 默认值
routers login name 路由名称 String 否 无
component 路由跳转页面 String 是 无
path 路由路径 String 是 无
meta 路由参数 String 是 无
main name 路由名称 String 是 无
component 路由跳转页面 String 是 无
path 路由路径 String 是 无
meta 路由参数 String 是 无
说明：框架内部集成了登录（login）和主页面（main）路由如果需自定义页面，传入以上参数指定相应页面

# stores

本节将介绍框架初始化中的路由配置

定义一个全局状态中心文件

const state = {
  loginToken: "",
};

const mutations = { // 设置变化
  ['setLoginToken'](state, token) {
    state.loginToken = token;
  },
};
const actions = {
  setLoginToken(context, token) {
    context.commit('setLoginToken', token);
  }
};
const getters = {
  loginToken: state => {
    return state.loginToken;
  }
};

export default {
  state,
  mutations,
  actions,
  getters,
}

定义引入全局状态中心index.js

// 批量导入状态管理模块
const files = require.context('./stores/modules', true, /.js$/);
const exports = {};
let keys = files.keys();
if (keys) {
  keys.map(filePathName => {
    let start = filePathName.lastIndexOf("/");
    let end = filePathName.lastIndexOf(".js");
    let fileName = filePathName.substring(start + 1, end);
    exports[fileName] = files(filePathName).default;
  });
}

export default exports;

说明：全局状态较多时，模块化管理更有利于后期维护和扩展。通过index.js统一导入
